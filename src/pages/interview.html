2、拍平数组，将一个多维数组转换成一个一维数组。

        const flatten = (array) =>{
            let flag = false
            let a = []
            array.forEach((item) => {
                if (Array.isArray(item)){
                    a = a.concat(item)
                    flag = true
                }else{
                    a.push(item)
                }
            })
            if (flag) return flatten(a) // 递归调用
            return a
        }

3、数组去重。

        const filterDuplicate = (array) => {
            const arr = []
            const hash = {}
            for(let i=0; i < array.length; i++){
                if (!hash[array[i]]) {
                    hash[array[i]] = true
                    arr.push(array[i])
                }
            }
            return arr
        }
